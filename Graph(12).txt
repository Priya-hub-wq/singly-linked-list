#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

class Edge {
public:
    int src, dest;
    int weight;
};

class DSU {
    vector<int> parent, rank;
public:
    DSU(int n) {
        parent.resize(n);
        rank.resize(n, 0);
        for(int i = 0; i < n; i++) parent[i] = i;
    }

    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]);
        return parent[x];
    }

    void unite(int x, int y) {
        int xroot = find(x);
        int yroot = find(y);

        if (xroot != yroot) {
            if (rank[xroot] < rank[yroot])
                parent[xroot] = yroot;
            else if (rank[xroot] > rank[yroot])
                parent[yroot] = xroot;
            else {
                parent[yroot] = xroot;
                rank[xroot]++;
            }
        }
    }
};

int main() {
    int V = 5;  // Houses
    vector<Edge> edges = {
        {0, 1, 4}, {0, 2, 2}, {1, 2, 1},
        {1, 3, 5}, {2, 3, 8}, {2, 4, 10}, {3, 4, 2}
    };

    // Sort edges by weight
    sort(edges.begin(), edges.end(), [](Edge a, Edge b){
        return a.weight < b.weight;
    });

    DSU dsu(V);
    vector<Edge> mst;
    int totalCost = 0;

    for (auto &e : edges) {
        if (dsu.find(e.src) != dsu.find(e.dest)) {
            dsu.unite(e.src, e.dest);
            mst.push_back(e);
            totalCost += e.weight;
        }
    }

    cout << "Minimum Pipeline Network (MST):\n";
    for (auto &e : mst)
        cout << "House " << e.src << " <--> House " << e.dest 
             << "  | Cost: " << e.weight << "\n";

    cout << "\nTotal Minimum Pipeline Cost: " << totalCost << endl;

    return 0;
}
