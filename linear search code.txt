#include <iostream>
using namespace std;

int main() {
    int n, key;

    // Input array size
    cout << "Enter number of elements: ";
    cin >> n;

    int arr[100]; // Adjust size if needed

    // Input array elements
    cout << "Enter " << n << " elements: ";
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Element to search
    cout << "Enter element to search: ";
    cin >> key;

    // Linear Search
    bool found = false;
    for (int i = 0; i < n; i++) {
        if (arr[i] == key) {
            cout << "Element found at position " << i + 1 << endl;
            found = true;
            break;
        }
    }

    if (!found) {
        cout << "Element not found in the array." << endl;
    }

    return 0;
}
